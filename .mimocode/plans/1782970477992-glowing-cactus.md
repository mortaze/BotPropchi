# Plan: Scheduled Group Messages (پیام‌های خودکار)

## Core Principle

This is a **standalone feature module** added alongside the existing Post system. The Post system remains 100% untouched — no model changes, no handler changes, no keyboard changes, no service changes. The new feature mirrors the Post UX using new models, new handlers, new keyboards, and new services, while reusing only **pure utility functions** that have no dependency on Post models.

## Step 1: Prisma Schema — New Models Only

### `prisma/schema.prisma` — ADD (do NOT modify existing models)

```prisma
model ScheduledMessage {
  id              Int                    @id @default(autoincrement())
  title           String
  slug            String?                @unique
  status          PostStatus             @default(DRAFT)  // reuse existing enum
  isPublished     Boolean                @default(false)
  createdAt       DateTime               @default(now())
  updatedAt       DateTime               @updatedAt

  // Scheduling
  intervalHours   Int?                   // 3, 6, 9, 12, 24, 168 (weekly)
  startTime       String?                // "HH:MM"
  targetChatId    BigInt?
  targetTopicId   BigInt?                // message_thread_id
  lastSentAt      DateTime?
  nextSendAt      DateTime?
  sendCount       Int                    @default(0)
  createdBy       BigInt?

  messages        ScheduledMessageMessage[]
  buttons         ScheduledMessageButton[]
  logs            ScheduledPostLog[]

  @@index([status])
  @@index([nextSendAt])
  @@index([isPublished])
  @@map("scheduled_messages")
}

model ScheduledMessageMessage {
  id              Int                @id @default(autoincrement())
  scheduledMessageId Int
  text            String             @default("")
  type            PostMessageType    @default(TEXT)  // reuse existing enum
  order           Int                @default(0)
  mediaFileId     String?
  caption         String?
  captionEntities Json?
  entities        Json?
  replyMarkup     Json?
  createdAt       DateTime           @default(now())
  updatedAt       DateTime           @updatedAt
  scheduledMessage ScheduledMessage  @relation(fields: [scheduledMessageId], references: [id], onDelete: Cascade)

  @@index([scheduledMessageId, order])
  @@map("scheduled_message_messages")
}

model ScheduledMessageButton {
  id              Int                @id @default(autoincrement())
  scheduledMessageId Int
  messageId       Int?
  row             Int                @default(0)
  col             Int                @default(0)
  text            String
  type            String             @default("URL")
  value           String?
  payload         Json?
  createdAt       DateTime           @default(now())
  scheduledMessage ScheduledMessage  @relation(fields: [scheduledMessageId], references: [id], onDelete: Cascade)

  @@index([scheduledMessageId])
  @@map("scheduled_message_buttons")
}

model ScheduledPostLog {
  id              Int       @id @default(autoincrement())
  scheduledMessageId Int
  sentAt          DateTime  @default(now())
  targetChatId    BigInt
  targetTopicId   BigInt?
  status          String    @default("SUCCESS")
  errorMessage    String?
  createdAt       DateTime  @default(now())
  scheduledMessage ScheduledMessage @relation(fields: [scheduledMessageId], references: [id], onDelete: Cascade)

  @@index([scheduledMessageId])
  @@index([sentAt])
  @@map("scheduled_post_logs")
}
```

**Enums reused (not modified):** `PostStatus`, `PostMessageType`, `PostButtonType` — these are existing enums in the schema, reused by reference.

Run `npm run db:push` after this step.

## Step 10: Forum Topic Support

### `src/services/group.service.ts` (MODIFY — +1 method)

```typescript
async fetchForumTopics(bot: Telegraf, chatId: number) {
  const chat = await bot.telegram.getChat(chatId);
  if (!chat.is_forum) return null;
  await prisma.telegramGroup.update({
    where: { chatId: BigInt(chatId) },
    data: { isForum: true, forumTopicsFetchedAt: new Date() },
  });
  return { isForum: true };
}
```
